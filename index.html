Caesar Cipher 🔐




def caesar(text, shift, mode): result = "" for char in text: if char.isalpha(): base = ord('a') if char.islower() else ord('A') shift_amount = shift if mode == "encode" else -shift result += chr((ord(char) - base + shift_amount) % 26 + base) else: result += char return result def run_cipher(): try: text = Element("text").value shift = int(Element("shift").value) mode = Element("mode").value result = caesar(text, shift, mode) Element("output").write("Result: " + result) except Exception as e: Element("output").write("Error: Please enter valid input")
